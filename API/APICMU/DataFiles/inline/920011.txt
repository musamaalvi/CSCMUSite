Label('isupper demo', 200, 20, size=20, bold=True)
Label('Enter any full name to get its initials', 200, 45, size=16)
Label('(This assumes that initials are in Upper Case)', 200, 65, size=16)

app.name = app.getTextInput('Enter your full name')
Label('Name: ' + app.name, 200, 180, size=20)
message = Label('Initials: ', 200, 220, size=20)

for c in app.name:
    if (c.isupper() == True):
        message.value += c
