Label('Bouncing motion demo', 200, 20, size=20)

c = Circle(200, 150, 20, fill='navy')
c.dx = 10

def onStep():
    c.centerX += c.dx
    if ((c.left < 0) or (c.right > 400)):
        # we went off the left edge or the right edge,
        # so reverse our direction ('bounce')
        c.dx = -c.dx
