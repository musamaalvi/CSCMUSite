Label('For Loop Demo #7', 200, 20, size=20, bold=True)
Label('Looping over shapes in a group', 200, 40, size=16)
Label('Click to add a small circle', 200, 60, size=16)
Label('Press any key to make the circles bigger', 200, 80, size=16)

circles = Group()

def onMousePress(x, y):
    circles.add(
        Circle(x, y, 10, fill=None, border='black')
    )

def onKeyPress(key):
    for circle in circles.children:
        circle.radius += 5
