Label('Group Methods Demo #1', 200, 20, size=20, bold=True)
Label('Click to add circles to the group', 200, 45, size=16)
Label('Press left and right arrows to move circles', 200, 65, size=16)

g = Group()

def onMousePress(x, y):
    g.add(Circle(x, y, 20, fill='cyan', border='black'))

def onKeyPress(key):
    if (key == 'right'):
        g.centerX += 10
    elif (key == 'left'):
        g.centerX -= 10
