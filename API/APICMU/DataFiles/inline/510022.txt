Line(200, 60, 200, 400, fill='lightGrey')
Line(0, 200, 400, 200, fill='lightGrey')

def onMousePress(x, y):
    if (x > 200):
        if (y > 200):
            Circle(x, y, 30, fill='lightGreen')
            Label('Yes!!!', x, y, size=18)
        else:
            Circle(x, y, 30, fill='orange')
            Label('nope', x, y, size=18)
