Label('Position Properties Example #2', 200, 20, size=20, bold=True)
Label('Move the right edge of the circled-star with the mouse', 200, 40, size=16)

g = Group(
      Circle(200, 200, 50, fill=None, border='black', borderWidth=5),
      Star(200, 200, 50, 5, fill=None, border='black', borderWidth=5)
    )

def onMouseMove(mouseX, mouseY):
    g.right = mouseX
    g.centerY = mouseY
