def drawBat(x, y):
    # body
    Polygon(x, y - 16, x + 6, y, x, y + 7, x - 6, y)

    # ears
    Label('X', x, y + 7, size=20)

    # eyes
    Circle(x - 3, y + 9, 1, fill='red')
    Circle(x + 3, y + 9, 1, fill='red')
